    def flag(cols):
        x = cols[0]
        j = cols[1]
        if x > j: return 1
        else: return 0
    train_calc['j']=j
    train_calc['flag'] = train_calc[['predicted','j']].apply(flag,axis=1)
    df_table = confusion_matrix(train_calc['Survived'],train_calc['flag'])

    r[i] = df_table[1,1] / (df_table[1,1] + df_table[1,0])
